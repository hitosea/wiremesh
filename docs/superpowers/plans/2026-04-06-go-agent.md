# WireMesh Go Agent Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the WireMesh Agent — a Go binary that runs on each VPN node, connects to the management platform via SSE, applies WireGuard/iptables configurations, and reports status.

**Architecture:** Single Go binary, no runtime dependencies. Reads config from `/etc/wiremesh/agent.yaml`. Connects to management platform via SSE for push notifications, pulls full config via HTTP GET, applies changes to local WireGuard interfaces and iptables rules, and periodically reports status via HTTP POST. All communication authenticated via per-node Bearer token.

**Tech Stack:** Go 1.22+, `gopkg.in/yaml.v3` (config parsing), standard library for HTTP/SSE/JSON, `os/exec` for system commands.

## API Contract (from implemented management platform)

The Go Agent is a **client** of these endpoints. All use `Authorization: Bearer {token}` header.

### GET /api/agent/sse
SSE stream. Events: `connected` (data: `{"nodeId":N}`), `peer_update`, `config_update`, `tunnel_update` (no data). Reconnect on disconnect.

### GET /api/agent/config
Returns JSON:
```json
{
  "data": {
    "node": { "id": 1, "name": "...", "ip": "...", "wgAddress": "10.0.0.1/24", "wgPort": 51820, "wgPrivateKey": "base64..." },
    "peers": [{ "publicKey": "base64...", "allowedIps": "10.0.0.100/32" }],
    "tunnels": {
      "interfaces": [{
        "name": "wm-tun1", "privateKey": "base64...", "address": "10.1.0.1/30",
        "listenPort": 51830, "peerPublicKey": "base64...",
        "peerAddress": "2.2.2.2", "peerPort": 51830, "role": "from"
      }],
      "iptablesRules": ["-A FORWARD -i wm-wg0 -o wm-tun1 -m comment --comment wm-line-3 -j ACCEPT"]
    },
    "xray": null,
    "version": "2026-04-06T10:00:00Z"
  }
}
```

### POST /api/agent/status
Body: `{ "is_online": true, "latency": 45, "transfers": [{"peer_public_key":"...","upload_bytes":N,"download_bytes":N}], "handshakes": [{"peer_public_key":"...","last_handshake":"ISO8601"}] }`

### POST /api/agent/error
Body: `{ "message": "error description" }`

### POST /api/agent/installed
Empty body OK. Marks node as online.

```go
package config

import (
	"fmt"
	"os"

	"gopkg.in/yaml.v3"
)

type Config struct {
	ServerURL      string `yaml:"server_url"`
	NodeID         int    `yaml:"node_id"`
	Token          string `yaml:"token"`
	ReportInterval int    `yaml:"report_interval"` // seconds, default 300
}

const DefaultConfigPath = "/etc/wiremesh/agent.yaml"

func Load(path string) (*Config, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read config: %w", err)
	}

	var cfg Config
	if err := yaml.Unmarshal(data, &cfg); err != nil {
		return nil, fmt.Errorf("parse config: %w", err)
	}

	if cfg.ServerURL == "" {
		return nil, fmt.Errorf("server_url is required")
	}
	if cfg.NodeID == 0 {
		return nil, fmt.Errorf("node_id is required")
	}
	if cfg.Token == "" {
		return nil, fmt.Errorf("token is required")
	}
	if cfg.ReportInterval <= 0 {
		cfg.ReportInterval = 300
	}

	return &cfg, nil
}
```

```go
package api

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"
)

type Client struct {
	baseURL    string
	token      string
	httpClient *http.Client
}

func NewClient(baseURL, token string) *Client {
	return &Client{
		baseURL: baseURL,
		token:   token,
		httpClient: &http.Client{
			Timeout: 30 * time.Second,
		},
	}
}

func (c *Client) doRequest(method, path string, body interface{}) (*http.Response, error) {
	var bodyReader io.Reader
	if body != nil {
		data, err := json.Marshal(body)
		if err != nil {
			return nil, fmt.Errorf("marshal body: %w", err)
		}
		bodyReader = bytes.NewReader(data)
	}

	url := c.baseURL + path
	req, err := http.NewRequest(method, url, bodyReader)
	if err != nil {
		return nil, fmt.Errorf("create request: %w", err)
	}

	req.Header.Set("Authorization", "Bearer "+c.token)
	if body != nil {
		req.Header.Set("Content-Type", "application/json")
	}

	return c.httpClient.Do(req)
}

// FetchConfig calls GET /api/agent/config and returns parsed config
func (c *Client) FetchConfig() (*ConfigData, error) {
	resp, err := c.doRequest("GET", "/api/agent/config", nil)
	if err != nil {
		return nil, fmt.Errorf("fetch config: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != 200 {
		body, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("fetch config: status %d: %s", resp.StatusCode, string(body))
	}

	var result ConfigResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("decode config: %w", err)
	}

	return &result.Data, nil
}

// ReportStatus calls POST /api/agent/status

	if resp.StatusCode != 200 {
		body, _ := io.ReadAll(resp.Body)
		return fmt.Errorf("report status: status %d: %s", resp.StatusCode, string(body))
	}

}
```

Create `agent/api/sse.go`:

```go
package api

import (
	"bufio"
	"context"
	"fmt"
	"log"
	"net/http"
	"strings"
	"time"
)

// SSEEvent represents a parsed Server-Sent Event
type SSEEvent struct {
	Event string
	Data  string
}

const (
	sseReconnectDelay    = 5 * time.Second
	sseMaxReconnectDelay = 60 * time.Second
)

// SSEClient connects to the SSE endpoint and delivers events via a channel.
// It automatically reconnects on disconnect with exponential backoff.
type SSEClient struct {
	client  *Client
	ctx     context.Context
	cancel  context.CancelFunc
	eventCh chan SSEEvent
}

// NewSSEClient creates a new SSE client. Call Start() to begin receiving events.
func NewSSEClient(client *Client) *SSEClient {
	ctx, cancel := context.WithCancel(context.Background())
	return &SSEClient{
		client:  client,
		ctx:     ctx,
		cancel:  cancel,
		eventCh: make(chan SSEEvent, 16),
	}
}

// Events returns the channel of SSE events
func (s *SSEClient) Events() <-chan SSEEvent {
	return s.eventCh
}

// Start begins the SSE connection loop in a goroutine. It reconnects on failure.
func (s *SSEClient) Start() {
	go s.connectLoop()
}

// Stop closes the SSE connection
func (s *SSEClient) Stop() {
	s.cancel()
}

func (s *SSEClient) connectLoop() {
	delay := sseReconnectDelay

	for {
		select {
		case <-s.ctx.Done():
			close(s.eventCh)
			return
		default:
		}

		err := s.connect()
		if err != nil {
			log.Printf("[SSE] Connection error: %v", err)
		}

		// Check if context was cancelled during connection
		select {
		case <-s.ctx.Done():
			close(s.eventCh)
			return
		default:
		}

		log.Printf("[SSE] Reconnecting in %v...", delay)
		select {
		case <-time.After(delay):
		case <-s.ctx.Done():
			close(s.eventCh)
			return
		}

		// Exponential backoff up to max
		delay = delay * 2
		if delay > sseMaxReconnectDelay {
			delay = sseMaxReconnectDelay
		}
	}
}

func (s *SSEClient) connect() error {
	url := s.client.baseURL + "/api/agent/sse"
	req, err := http.NewRequestWithContext(s.ctx, "GET", url, nil)
	if err != nil {
		return fmt.Errorf("create request: %w", err)
	}

	req.Header.Set("Authorization", "Bearer "+s.client.token)
	req.Header.Set("Accept", "text/event-stream")

	// Use a client with no timeout for SSE (long-lived connection)
	httpClient := &http.Client{Timeout: 0}
	resp, err := httpClient.Do(req)
	if err != nil {
		return fmt.Errorf("connect: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != 200 {
		return fmt.Errorf("unexpected status: %d", resp.StatusCode)
	}

	log.Println("[SSE] Connected to management platform")

	// Reset backoff on successful connection
	scanner := bufio.NewScanner(resp.Body)
	var currentEvent SSEEvent

	for scanner.Scan() {
		line := scanner.Text()

		if line == "" {
			// Empty line = event boundary
			if currentEvent.Event != "" {
				select {
				case s.eventCh <- currentEvent:
				case <-s.ctx.Done():
					return nil
				}
			}
			currentEvent = SSEEvent{}
			continue
		}

		if strings.HasPrefix(line, "event: ") {
			currentEvent.Event = strings.TrimPrefix(line, "event: ")
		} else if strings.HasPrefix(line, "data: ") {
			currentEvent.Data = strings.TrimPrefix(line, "data: ")
		}
	}

	if err := scanner.Err(); err != nil {
		return fmt.Errorf("read stream: %w", err)
	}

	return fmt.Errorf("stream closed by server")
}
```

```go
		time.Sleep(100 * time.Millisecond)
	}))

	sse := NewSSEClient(client)
	sse.Start()

	var events []SSEEvent
	timeout := time.After(2 * time.Second)
```

```go
package wg

import (
	"fmt"
	"log"
```
```go
	"os"
	"strings"

	"github.com/wiremesh/agent/api"
```

```go
// ActiveTunnel tracks a currently active tunnel interface
type ActiveTunnel struct {
	Name       string
	PrivateKey string
	Address    string
	ListenPort int
}

// SyncTunnels compares desired tunnel interfaces with current active ones.
// Creates new, updates existing, and destroys removed tunnels.
// Returns the new set of active tunnels.
func SyncTunnels(desired []api.TunnelInterface, active map[string]ActiveTunnel) (map[string]ActiveTunnel, error) {
	desiredMap := make(map[string]api.TunnelInterface)
	for _, iface := range desired {
		desiredMap[iface.Name] = iface
	}

	var errors []string

	// Destroy tunnels that are no longer in the desired list
	for name := range active {
		if _, exists := desiredMap[name]; !exists {
			log.Printf("[tunnel] Destroying %s", name)
			if err := destroyTunnel(name); err != nil {
				errors = append(errors, fmt.Sprintf("destroy %s: %v", name, err))
			}
			delete(active, name)
		}
	}

	// Create or update tunnels
	for _, iface := range desired {
		existing, exists := active[iface.Name]

		if !exists {
			// New tunnel — create
			log.Printf("[tunnel] Creating %s", iface.Name)
			if err := createTunnel(iface); err != nil {
				errors = append(errors, fmt.Sprintf("create %s: %v", iface.Name, err))
				continue
			}
		} else if tunnelChanged(existing, iface) {
			// Existing tunnel with changed config — update via syncconf
			log.Printf("[tunnel] Updating %s", iface.Name)
			if err := updateTunnel(iface); err != nil {
				errors = append(errors, fmt.Sprintf("update %s: %v", iface.Name, err))
				continue
			}
		} else {
			// No change
			continue
		}

		active[iface.Name] = ActiveTunnel{
			Name:       iface.Name,
			PrivateKey: iface.PrivateKey,
			Address:    iface.Address,
			ListenPort: iface.ListenPort,
		}
	}

	if len(errors) > 0 {
		return active, fmt.Errorf("tunnel sync errors: %s", strings.Join(errors, "; "))
	}
	return active, nil
}

func tunnelChanged(active ActiveTunnel, desired api.TunnelInterface) bool {
	return active.PrivateKey != desired.PrivateKey ||
		active.Address != desired.Address ||
		active.ListenPort != desired.ListenPort
}

func createTunnel(iface api.TunnelInterface) error {
	confPath := writeTunnelConf(iface)

	// 1. Create WireGuard interface
	if err := IpLinkAdd(iface.Name); err != nil {
		return fmt.Errorf("ip link add: %w", err)
	}

	// 2. Apply config
	if err := WgSetConf(iface.Name, confPath); err != nil {
		IpLinkDel(iface.Name) // cleanup on failure
		return fmt.Errorf("wg setconf: %w", err)
	}

	// 3. Assign address
	if err := IpAddrAdd(iface.Address, iface.Name); err != nil {
		IpLinkDel(iface.Name)
		return fmt.Errorf("ip addr add: %w", err)
	}

	// 4. Bring up
	if err := IpLinkSetUp(iface.Name); err != nil {
		IpLinkDel(iface.Name)
		return fmt.Errorf("ip link set up: %w", err)
	}

	return nil
}

func updateTunnel(iface api.TunnelInterface) error {
	confPath := writeTunnelConf(iface)
	return WgSyncConf(iface.Name, confPath)
}

func destroyTunnel(name string) error {
	// Bring down and delete
	IpLinkSetDown(name) // ignore error — may already be down
	if err := IpLinkDel(name); err != nil {
		return err
	}

	// Remove config file
	confPath := fmt.Sprintf("%s/%s.conf", WgConfigDir, name)
	os.Remove(confPath) // ignore error — file may not exist

	return nil
}

func writeTunnelConf(iface api.TunnelInterface) string {
	confPath := fmt.Sprintf("%s/%s.conf", WgConfigDir, iface.Name)

	var sb strings.Builder
	sb.WriteString("[Interface]\n")
	sb.WriteString(fmt.Sprintf("PrivateKey = %s\n", iface.PrivateKey))
	sb.WriteString(fmt.Sprintf("ListenPort = %d\n", iface.ListenPort))
	sb.WriteString("\n[Peer]\n")
	sb.WriteString(fmt.Sprintf("PublicKey = %s\n", iface.PeerPublicKey))

	// AllowedIPs: for downstream tunnels use 0.0.0.0/0, for upstream use the peer's /30
	if iface.Role == "from" {
		sb.WriteString("AllowedIPs = 0.0.0.0/0\n")
	} else {
		// Extract /30 subnet from peer address
		sb.WriteString(fmt.Sprintf("AllowedIPs = %s\n", peerAllowedIPs(iface.PeerPublicKey, iface)))
	}

	sb.WriteString(fmt.Sprintf("Endpoint = %s:%d\n", iface.PeerAddress, iface.PeerPort))
	sb.WriteString("PersistentKeepalive = 25\n")

	os.MkdirAll(WgConfigDir, 0700)
	os.WriteFile(confPath, []byte(sb.String()), 0600)

	return confPath
}

// peerAllowedIPs determines the AllowedIPs for a tunnel peer.
// "from" role tunnels (downstream) use 0.0.0.0/0 to accept all traffic.
// "to" role tunnels (upstream) use the peer's tunnel subnet.
func peerAllowedIPs(_ string, iface api.TunnelInterface) string {
	// For upstream (to role), allow the /30 subnet
	// Extract base from address like "10.1.0.2/30" → "10.1.0.0/30"
	parts := strings.Split(iface.Address, "/")
	if len(parts) == 2 {
		ipParts := strings.Split(parts[0], ".")
		if len(ipParts) == 4 {
			// /30 network: clear last 2 bits
			lastOctet := 0
			fmt.Sscanf(ipParts[3], "%d", &lastOctet)
			base := lastOctet & 0xFC // clear last 2 bits
			return fmt.Sprintf("%s.%s.%s.%d/%s", ipParts[0], ipParts[1], ipParts[2], base, parts[1])
		}
	}
	return "0.0.0.0/0"
}
```

```go
func SyncRules(desiredRules []string) error {
	// 1. Find all existing WireMesh iptables rules
	existingRules, err := listWireMeshRules()
	if err != nil {
		log.Printf("[iptables] Warning: failed to list existing rules: %v", err)
	}

	// 2. Remove rules that are no longer desired
	desiredSet := make(map[string]bool)
	for _, rule := range desiredRules {
		desiredSet[normalizeRule(rule)] = true
	}

	for _, rule := range existingRules {
		if !desiredSet[normalizeRule(rule)] {
			log.Printf("[iptables] Removing: iptables %s", rule)
			removeRule(rule)
		}
	}

	// 3. Add rules that don't already exist
	existingSet := make(map[string]bool)
	for _, rule := range existingRules {
		existingSet[normalizeRule(rule)] = true
	}

	for _, rule := range desiredRules {
		if !existingSet[normalizeRule(rule)] {
			log.Printf("[iptables] Adding: iptables %s", rule)
			if err := addRule(rule); err != nil {
				log.Printf("[iptables] Error adding rule: %v", err)
			}
		}
	}

	return nil
}

// RemoveAllWireMeshRules removes all iptables rules with "wm-" comments
func RemoveAllWireMeshRules() error {
	rules, err := listWireMeshRules()
	if err != nil {
		return err
	}
	for _, rule := range rules {
		removeRule(rule)
	}
	return nil
}

// listWireMeshRules finds all iptables rules that have a "wm-" comment
func listWireMeshRules() ([]string, error) {
	var allRules []string

	// Check FORWARD chain in filter table
	forwardRules, err := listChainRules("filter", "FORWARD")
	if err == nil {
		allRules = append(allRules, forwardRules...)
	}

	// Check POSTROUTING chain in nat table
	natRules, err := listChainRules("nat", "POSTROUTING")
	if err == nil {
		allRules = append(allRules, natRules...)
	}

	return allRules, nil
}

func listChainRules(table, chain string) ([]string, error) {
	args := []string{"-t", table, "-S", chain}
	output, err := exec.Command("iptables", args...).CombinedOutput()
	if err != nil {
		return nil, fmt.Errorf("iptables %s: %w", strings.Join(args, " "), err)
	}

	var wmRules []string
	for _, line := range strings.Split(string(output), "\n") {
		line = strings.TrimSpace(line)
		if strings.Contains(line, "wm-") && strings.HasPrefix(line, "-A ") {
			wmRules = append(wmRules, line)
		}
	}
	return wmRules, nil
}

func addRule(rule string) error {
	// Rule comes as "-A FORWARD ..." or "-t nat -A POSTROUTING ..."
	args := strings.Fields(rule)
	cmd := exec.Command("iptables", args...)
	output, err := cmd.CombinedOutput()
	if err != nil {
		return fmt.Errorf("iptables %s: %w: %s", rule, err, string(output))
	}
	return nil
}

func removeRule(rule string) {
	// Convert -A to -D for deletion
	deleteRule := strings.Replace(rule, "-A ", "-D ", 1)
	args := strings.Fields(deleteRule)
	exec.Command("iptables", args...).CombinedOutput() // ignore errors
}

func normalizeRule(rule string) string {
	return strings.TrimSpace(rule)
}
```

```go
// Collect gathers node status: transfers, handshakes, and latency.
func Collect(serverURL string) *api.StatusReport {
	report := &api.StatusReport{
		IsOnline: true,
	}

	// 1. Measure latency (ping management platform)
	latency := measureLatency(serverURL)
	if latency >= 0 {
		report.Latency = &latency
	}

	// 2. Collect WireGuard transfer stats from all interfaces
	transfers := collectTransfers()
	if len(transfers) > 0 {
		report.Transfers = transfers
	}

	// 3. Collect handshake data from wm-wg0
	handshakes := collectHandshakes()
	if len(handshakes) > 0 {
		report.Handshakes = handshakes
	}

	return report
}

func measureLatency(serverURL string) int {
	start := time.Now()
	client := &http.Client{Timeout: 10 * time.Second}
	resp, err := client.Head(serverURL)
	if err != nil {
		return -1
	}
	resp.Body.Close()
	return int(time.Since(start).Milliseconds())
}

// collectTransfers runs "wg show <iface> transfer" for all wm- interfaces
func collectTransfers() []api.TransferReport {
	var transfers []api.TransferReport

	// Get transfer data from wm-wg0
	transfers = append(transfers, collectInterfaceTransfers(wg.MainInterface)...)

	// Get transfer data from tunnel interfaces
	output, err := wg.RunSilent("ip", "-o", "link", "show")
	if err != nil {
		return transfers
	}

	for _, line := range strings.Split(output, "\n") {
		// Look for wm-tun interfaces
		fields := strings.Fields(line)
		for i, f := range fields {
			if i > 0 && strings.HasPrefix(f, "wm-tun") {
				name := strings.TrimSuffix(f, ":")
				transfers = append(transfers, collectInterfaceTransfers(name)...)
				break
			}
		}
	}

	return transfers
}

func collectInterfaceTransfers(iface string) []api.TransferReport {
	output, err := wg.WgShow(iface, "transfer")
	if err != nil {
		return nil
	}

	var transfers []api.TransferReport
	for _, line := range strings.Split(output, "\n") {
		line = strings.TrimSpace(line)
		if line == "" {
			continue
		}
		// Format: <peer_public_key>\t<rx_bytes>\t<tx_bytes>
		parts := strings.Split(line, "\t")
		if len(parts) != 3 {
			continue
		}

		rx, _ := strconv.ParseInt(parts[1], 10, 64)
		tx, _ := strconv.ParseInt(parts[2], 10, 64)

		transfers = append(transfers, api.TransferReport{
			PeerPublicKey: parts[0],
			UploadBytes:   tx,
			DownloadBytes: rx,
		})
	}
	return transfers
}

// collectHandshakes runs "wg show wm-wg0 latest-handshakes"
func collectHandshakes() []api.HandshakeReport {
	output, err := wg.WgShow(wg.MainInterface, "latest-handshakes")
	if err != nil {
		return nil
	}

	var handshakes []api.HandshakeReport
	for _, line := range strings.Split(output, "\n") {
		line = strings.TrimSpace(line)
		if line == "" {
			continue
		}
		// Format: <peer_public_key>\t<unix_timestamp>
		parts := strings.Split(line, "\t")
		if len(parts) != 2 {
			continue
		}

		timestamp, err := strconv.ParseInt(parts[1], 10, 64)
		if err != nil || timestamp == 0 {
			continue // 0 means no handshake yet
		}

		t := time.Unix(timestamp, 0).UTC()
		handshakes = append(handshakes, api.HandshakeReport{
			PeerPublicKey: parts[0],
			LastHandshake: t.Format(time.RFC3339),
		})
	}
	return handshakes
}

// FormatBytes converts bytes to human-readable string (for logging)
func FormatBytes(bytes int64) string {
	const unit = 1024
	if bytes < unit {
		return fmt.Sprintf("%d B", bytes)
	}
	div, exp := int64(unit), 0
	for n := bytes / unit; n >= unit; n /= unit {
		div *= unit
		exp++
	}
	return fmt.Sprintf("%.1f %ciB", float64(bytes)/float64(div), "KMGTPE"[exp])
}
```

```go
import (
	"context"
	"log"

	"github.com/wiremesh/agent/collector"
	"github.com/wiremesh/agent/config"

type Agent struct {
	cfg            *config.Config
	client         *api.Client
	sse            *api.SSEClient
	activeTunnels  map[string]wg.ActiveTunnel
	lastVersion    string
	ctx            context.Context
	cancel         context.CancelFunc
}

func New(cfg *config.Config) *Agent {
	ctx, cancel := context.WithCancel(context.Background())
	client := api.NewClient(cfg.ServerURL, cfg.Token)

	return &Agent{
		cfg:           cfg,
		client:        client,
		activeTunnels: make(map[string]wg.ActiveTunnel),
		ctx:           ctx,
		cancel:        cancel,
	}
}

// Run starts the agent. Blocks until context is cancelled.
func (a *Agent) Run() error {
	// 1. Report installation complete
	log.Println("[agent] Reporting installation complete...")
	if err := a.client.ReportInstalled(); err != nil {
		log.Printf("[agent] Warning: failed to report installed: %v", err)
	}

	// 2. Initial config pull and apply
	log.Println("[agent] Pulling initial configuration...")
	if err := a.pullAndApplyConfig(); err != nil {
		log.Printf("[agent] Warning: initial config pull failed: %v", err)
		a.client.ReportError("Initial config pull failed: " + err.Error())
	}

	// 3. Start SSE connection
	a.sse = api.NewSSEClient(a.client)
	a.sse.Start()
	log.Println("[agent] SSE connection started")

	// 4. Start status reporting ticker
	reportTicker := time.NewTicker(time.Duration(a.cfg.ReportInterval) * time.Second)
	defer reportTicker.Stop()

	// 5. Event loop
	for {
		select {
		case <-a.ctx.Done():
			a.shutdown()
			return nil

		case evt, ok := <-a.sse.Events():
			if !ok {
				log.Println("[agent] SSE event channel closed")
				return nil
			}
			a.handleSSEEvent(evt)

		case <-reportTicker.C:
			a.reportStatus()
		}
	}
}

// Stop signals the agent to shut down
func (a *Agent) Stop() {
	a.cancel()
}

func (a *Agent) handleSSEEvent(evt api.SSEEvent) {
	log.Printf("[agent] SSE event: %s", evt.Event)

	switch evt.Event {
	case "connected":
		log.Println("[agent] SSE connected to management platform")
	case "peer_update", "config_update", "tunnel_update":
		// All three events trigger a full config pull
		if err := a.pullAndApplyConfig(); err != nil {
			log.Printf("[agent] Config apply failed: %v", err)
			a.client.ReportError("Config apply failed: " + err.Error())
		}
	default:
		log.Printf("[agent] Unknown SSE event: %s", evt.Event)
	}
}

func (a *Agent) pullAndApplyConfig() error {
	cfgData, err := a.client.FetchConfig()
	if err != nil {
		return err
	}

	// Check version — skip if unchanged
	if cfgData.Version == a.lastVersion && a.lastVersion != "" {
		log.Println("[agent] Config version unchanged, skipping")
		return nil
	}

	log.Printf("[agent] Applying config version %s", cfgData.Version)

	// 1. Sync wm-wg0 peers
	if err := wg.SyncMainInterface(cfgData.Node, cfgData.Peers); err != nil {
		return err
	}

	// 2. Sync tunnel interfaces
	newActive, err := wg.SyncTunnels(cfgData.Tunnels.Interfaces, a.activeTunnels)
	if err != nil {
		log.Printf("[agent] Tunnel sync had errors: %v", err)
		// Continue — partial success is better than stopping
	}
	a.activeTunnels = newActive

	// 3. Sync iptables rules
	if err := iptables.SyncRules(cfgData.Tunnels.IptablesRules); err != nil {
		log.Printf("[agent] iptables sync error: %v", err)
	}

	// 4. TODO: Xray config update (future enhancement)

	a.lastVersion = cfgData.Version
	log.Printf("[agent] Config applied successfully. Tunnels: %d, iptables rules: %d",
		len(a.activeTunnels), len(cfgData.Tunnels.IptablesRules))

	return nil
}

func (a *Agent) reportStatus() {
	report := collector.Collect(a.cfg.ServerURL)

	if err := a.client.ReportStatus(report); err != nil {
		log.Printf("[agent] Status report failed: %v", err)
	} else {
		log.Printf("[agent] Status reported (latency: %v, transfers: %d, handshakes: %d)",
			report.Latency, len(report.Transfers), len(report.Handshakes))
	}
}

func (a *Agent) shutdown() {
	log.Println("[agent] Shutting down...")

	// Stop SSE
	if a.sse != nil {
		a.sse.Stop()
	}

	// Clean up tunnel interfaces
	for name := range a.activeTunnels {
		log.Printf("[agent] Destroying tunnel %s", name)
		wg.IpLinkSetDown(name)
		wg.IpLinkDel(name)
	}

	// Remove WireMesh iptables rules
	iptables.RemoveAllWireMeshRules()

	log.Println("[agent] Shutdown complete")
}
```

```go
package main

import (

	}()
```
